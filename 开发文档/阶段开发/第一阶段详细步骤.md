```kotlin
// WorkflowModels.kt
data class WorkflowNode(
    val id: String,
    val type: NodeType,
    val title: String,
    val x: Float,
    val y: Float,
    val inputs: List<NodeInput> = emptyList(),
    val outputs: List<NodeOutput> = emptyList(),
    val config: Map<String, Any> = emptyMap()
)

data class NodeInput(
    val id: String,
    val name: String,
    val type: String,
    val required: Boolean = false
)

data class NodeOutput(
    val id: String,
    val name: String,
    val type: String
)

data class WorkflowConnection(
    val id: String,
    val sourceNodeId: String,
    val sourceOutputId: String,
    val targetNodeId: String,
    val targetInputId: String
)

data class Workflow(
    val id: String,
    val name: String,
    val description: String = "",
    val nodes: List<WorkflowNode>,
    val connections: List<WorkflowConnection>,
    val version: String = "1.0",
    val createdAt: Long = System.currentTimeMillis(),
    val updatedAt: Long = System.currentTimeMillis()
)

enum class NodeType(val displayName: String) {
    START("开始"),
    END("结束"),
    HTTP_REQUEST("HTTP请求"),
    DATA_TRANSFORM("数据转换"),
    CONDITION("条件判断"),
    LOOP("循环"),
    DELAY("延时"),
    SCRIPT("脚本执行")
}
```

### Day 2: ViewModel和状态管理

#### 2.1 创建WorkflowViewModel
```kotlin
class WorkflowViewModel : ViewModel() {
    private val _workflow = MutableStateFlow(createEmptyWorkflow())
    val workflow: StateFlow<Workflow> = _workflow

    private val _selectedNodeId = MutableStateFlow<String?>(null)
    val selectedNodeId: StateFlow<String?> = _selectedNodeId

    fun addNode(type: NodeType, x: Float, y: Float) {
        val newNode = WorkflowNode(
            id = UUID.randomUUID().toString(),
            type = type,
            title = type.displayName,
            x = x,
            y = y,
            inputs = getDefaultInputs(type),
            outputs = getDefaultOutputs(type)
        )
        
        _workflow.value = _workflow.value.copy(
            nodes = _workflow.value.nodes + newNode,
            updatedAt = System.currentTimeMillis()
        )
    }

    fun moveNode(nodeId: String, deltaX: Float, deltaY: Float) {
        _workflow.value = _workflow.value.copy(
            nodes = _workflow.value.nodes.map { node ->
                if (node.id == nodeId) {
                    node.copy(x = node.x + deltaX, y = node.y + deltaY)
                } else node
            },
            updatedAt = System.currentTimeMillis()
        )
    }

    fun deleteNode(nodeId: String) {
        _workflow.value = _workflow.value.copy(
            nodes = _workflow.value.nodes.filter { it.id != nodeId },
            connections = _workflow.value.connections.filter { 
                it.sourceNodeId != nodeId && it.targetNodeId != nodeId 
            },
            updatedAt = System.currentTimeMillis()
        )
    }

    fun selectNode(nodeId: String?) {
        _selectedNodeId.value = nodeId
    }

    private fun createEmptyWorkflow() = Workflow(
        id = UUID.randomUUID().toString(),
        name = "新建工作流",
        nodes = emptyList(),
        connections = emptyList()
    )
}
```

### Day 3: 基础UI组件

#### 3.1 创建节点视图组件
```kotlin
@Composable
fun WorkflowNodeView(
    node: WorkflowNode,
    isSelected: Boolean,
    onMove: (Float, Float) -> Unit,
    onSelect: () -> Unit,
    onDelete: () -> Unit,
    modifier: Modifier = Modifier
) {
    var isDragging by remember { mutableStateOf(false) }

    Card(
        modifier = modifier
            .size(160.dp, 100.dp)
            .pointerInput(Unit) {
                detectDragGestures(
                    onDragStart = { isDragging = true },
                    onDragEnd = { isDragging = false }
                ) { _, dragAmount ->
                    onMove(dragAmount.x, dragAmount.y)
                }
            }
            .clickable { onSelect() },
        colors = CardDefaults.cardColors(
            containerColor = if (isSelected) 
                MaterialTheme.colorScheme.primaryContainer
            else MaterialTheme.colorScheme.surface
        ),
        border = if (isSelected) 
            BorderStroke(2.dp, MaterialTheme.colorScheme.primary) 
        else null,
        elevation = CardDefaults.cardElevation(
            defaultElevation = if (isDragging) 8.dp else 4.dp
        )
    ) {
        // 节点内容布局
        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(8.dp),
            verticalArrangement = Arrangement.SpaceBetween
        ) {
            // 标题和删除按钮
            Row(
                modifier = Modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.SpaceBetween,
                verticalAlignment = Alignment.CenterVertically
            ) {
                Icon(
                    imageVector = getNodeIcon(node.type),
                    contentDescription = null,
                    tint = getNodeColor(node.type),
                    modifier = Modifier.size(20.dp)
                )
                
                if (isSelected) {
                    IconButton(
                        onClick = onDelete,
                        modifier = Modifier.size(20.dp)
                    ) {
                        Icon(
                            Icons.Default.Close,
                            contentDescription = "删除",
                            modifier = Modifier.size(16.dp)
                        )
                    }
                }
            }
            
            Text(
                text = node.title,
                fontSize = 12.sp,
                fontWeight = FontWeight.Medium,
                maxLines = 2
            )
            
            // 输入输出端口
            Row(
                modifier = Modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.SpaceBetween
            ) {
                // 输入端口
                Column {
                    node.inputs.take(2).forEach { _ ->
                        Box(
                            modifier = Modifier
                                .size(8.dp)
                                .clip(CircleShape)
                                .background(MaterialTheme.colorScheme.secondary)
                        )
                        Spacer(modifier = Modifier.height(4.dp))
                    }
                }
                
                // 输出端口
                Column {
                    node.outputs.take(2).forEach { _ ->
                        Box(
                            modifier = Modifier
                                .size(8.dp)
                                .clip(CircleShape)
                                .background(MaterialTheme.colorScheme.tertiary)
                        )
                        Spacer(modifier = Modifier.height(4.dp))
                    }
                }
            }
        }
    }
}
```

### Day 4: 主编辑器界面

#### 4.1 创建WorkflowEditor
```kotlin
@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun WorkflowEditor(
    viewModel: WorkflowViewModel = viewModel(),
    modifier: Modifier = Modifier
) {
    val workflow by viewModel.workflow.collectAsState()
    val selectedNodeId by viewModel.selectedNodeId.collectAsState()
    val density = LocalDensity.current
    
    var showNodePalette by remember { mutableStateOf(false) }

    Column(modifier = modifier.fillMaxSize()) {
        // 工具栏
        TopAppBar(
            title = { Text(workflow.name) },
            actions = {
                IconButton(onClick = { showNodePalette = true }) {
                    Icon(Icons.Default.Add, "添加节点")
                }
            }
        )

        // 画布区域
        Box(
            modifier = Modifier
                .fillMaxSize()
                .background(Color(0xFFF5F5F5))
                .pointerInput(Unit) {
                    detectTapGestures { 
                        viewModel.selectNode(null)
                    }
                }
        ) {
            // 渲染节点
            workflow.nodes.forEach { node ->
                WorkflowNodeView(
                    node = node,
                    isSelected = node.id == selectedNodeId,
                    onMove = { deltaX, deltaY ->
                        viewModel.moveNode(node.id, deltaX, deltaY)
                    },
                    onSelect = { viewModel.selectNode(node.id) },
                    onDelete = { viewModel.deleteNode(node.id) },
                    modifier = Modifier.offset(
                        x = with(density) { node.x.dp },
                        y = with(density) { node.y.dp }
                    )
                )
            }
        }
    }

    // 节点选择面板
    if (showNodePalette) {
        NodePaletteDialog(
            onDismiss = { showNodePalette = false },
            onNodeSelected = { nodeType ->
                viewModel.addNode(nodeType, 100f, 100f)
                showNodePalette = false
            }
        )
    }
}
```

```kotlin
@Composable
fun NodePaletteDialog(
    onDismiss: () -> Unit,
    onNodeSelected: (NodeType) -> Unit
) {
    AlertDialog(
        onDismissRequest = onDismiss,
        title = { Text("选择节点类型") },
        text = {
            LazyColumn {
                items(NodeType.values()) { nodeType ->
                    Card(
                        modifier = Modifier
                            .fillMaxWidth()
                            .padding(vertical = 4.dp),
                        colors = CardDefaults.cardColors(
                            containerColor = MaterialTheme.colorScheme.surfaceVariant
                        ),
                        onClick = { onNodeSelected(nodeType) }
                    ) {
                        Row(
                            modifier = Modifier.padding(16.dp),
                            verticalAlignment = Alignment.CenterVertically
                        ) {
                            Icon(
                                imageVector = getNodeIcon(nodeType),
                                contentDescription = null,
                                modifier = Modifier.size(24.dp)
                            )
                            Spacer(modifier = Modifier.width(12.dp))
                            Text(
                                text = nodeType.displayName,
                                fontWeight = FontWeight.Medium
                            )
                        }
                    }
                }
            }
        },
        confirmButton = {
            TextButton(onClick = onDismiss) {
                Text("取消")
            }
        }
    )
}
```

```kotlin
private fun getNodeIcon(type: NodeType) = when (type) {
    NodeType.START -> Icons.Default.PlayArrow
    NodeType.END -> Icons.Default.Stop
    NodeType.HTTP_REQUEST -> Icons.Default.Http
    NodeType.DATA_TRANSFORM -> Icons.Default.Transform
    NodeType.CONDITION -> Icons.Default.QuestionMark
    NodeType.LOOP -> Icons.Default.Loop
    NodeType.DELAY -> Icons.Default.Timer
    NodeType.SCRIPT -> Icons.Default.Code
}

private fun getNodeColor(type: NodeType) = when (type) {
    NodeType.START -> Color(0xFF4CAF50)
    NodeType.END -> Color(0xFFF44336)
    NodeType.HTTP_REQUEST -> Color(0xFF2196F3)
    NodeType.DATA_TRANSFORM -> Color(0xFFFF9800)
    NodeType.CONDITION -> Color(0xFF9C27B0)
    NodeType.LOOP -> Color(0xFF607D8B)
    NodeType.DELAY -> Color(0xFFFFEB3B)
    NodeType.SCRIPT -> Color(0xFF795548)
}
```

## 第一阶段验收标准

### 功能检查清单
- [ ] 可以添加不同类型的节点
- [ ] 节点可以拖拽移动
- [ ] 节点可以选择和删除
- [ ] 界面响应流畅，无卡顿
- [ ] 节点图标和颜色正确显示
- [ ] 工具栏功能正常

### 测试步骤
1. 启动应用，查看空白画布
2. 点击"+"按钮，选择不同节点类型
3. 拖拽节点到不同位置
4. 点击节点进行选择
5. 删除选中的节点
6. 重复操作，确保稳定性

### 性能要求
- 应用启动时间 < 3秒
- 节点拖拽响应延迟 < 100ms
- 支持同时显示 50+ 节点不卡顿

完成第一阶段后，即可进入第二阶段的配置化系统开发。
