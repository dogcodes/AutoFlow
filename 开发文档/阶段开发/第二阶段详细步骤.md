```kotlin
class WorkflowViewModel : ViewModel() {
    private val gson = Gson()
    private val httpClient = OkHttpClient()

    fun exportToJson(): String = gson.toJson(_workflow.value)

    fun importFromJson(json: String): Boolean {
        return try {
            val workflow = gson.fromJson(json, Workflow::class.java)
            _workflow.value = workflow.copy(updatedAt = System.currentTimeMillis())
            true
        } catch (e: Exception) {
            false
        }
    }

    fun loadFromUrl(url: String, onResult: (Boolean, String?) -> Unit) {
        viewModelScope.launch {
            val request = Request.Builder().url(url).build()
            
            httpClient.newCall(request).enqueue(object : Callback {
                override fun onFailure(call: Call, e: IOException) {
                    onResult(false, e.message)
                }

                override fun onResponse(call: Call, response: Response) {
                    response.body?.string()?.let { json ->
                        val success = importFromJson(json)
                        onResult(success, if (success) null else "解析失败")
                    } ?: onResult(false, "响应为空")
                }
            })
        }
    }

    fun loadSampleWorkflow() {
        val sampleWorkflow = Workflow(
            id = "sample-workflow",
            name = "示例工作流",
            description = "获取GitHub用户信息的示例工作流",
            nodes = listOf(
                WorkflowNode(
                    id = "start",
                    type = NodeType.START,
                    title = "开始",
                    x = 50f, y = 100f,
                    outputs = listOf(NodeOutput("out", "输出", "any"))
                ),
                WorkflowNode(
                    id = "http",
                    type = NodeType.HTTP_REQUEST,
                    title = "获取GitHub用户",
                    x = 250f, y = 100f,
                    inputs = listOf(NodeInput("in", "输入", "any", true)),
                    outputs = listOf(NodeOutput("response", "响应", "object")),
                    config = mapOf(
                        "url" to "https://api.github.com/users/octocat",
                        "method" to "GET"
                    )
                ),
                WorkflowNode(
                    id = "end",
                    type = NodeType.END,
                    title = "结束",
                    x = 450f, y = 100f,
                    inputs = listOf(NodeInput("in", "输入", "any"))
                )
            ),
            connections = listOf(
                WorkflowConnection("c1", "start", "out", "http", "in"),
                WorkflowConnection("c2", "http", "response", "end", "in")
            )
        )
        
        _workflow.value = sampleWorkflow
    }
}
```

```kotlin
fun addConnection(sourceNodeId: String, sourceOutputId: String, 
                 targetNodeId: String, targetInputId: String) {
    val connection = WorkflowConnection(
        id = UUID.randomUUID().toString(),
        sourceNodeId = sourceNodeId,
        sourceOutputId = sourceOutputId,
        targetNodeId = targetNodeId,
        targetInputId = targetInputId
    )
    
    _workflow.value = _workflow.value.copy(
        connections = _workflow.value.connections + connection,
        updatedAt = System.currentTimeMillis()
    )
}

fun removeConnection(connectionId: String) {
    _workflow.value = _workflow.value.copy(
        connections = _workflow.value.connections.filter { it.id != connectionId },
        updatedAt = System.currentTimeMillis()
    )
}
```

### 8.3 连接线绘制
```kotlin
@Composable
fun ConnectionLine(
    startX: Float,
    startY: Float,
    endX: Float,
    endY: Float,
    modifier: Modifier = Modifier
) {
    Canvas(modifier = modifier.fillMaxSize()) {
        drawConnection(
            start = Offset(startX, startY),
            end = Offset(endX, endY)
        )
    }
}

private fun DrawScope.drawConnection(start: Offset, end: Offset) {
    // 贝塞尔曲线连接
    val controlPoint1 = Offset(start.x + (end.x - start.x) * 0.5f, start.y)
    val controlPoint2 = Offset(start.x + (end.x - start.x) * 0.5f, end.y)
    
    val path = Path().apply {
        moveTo(start.x, start.y)
        cubicTo(
            controlPoint1.x, controlPoint1.y,
            controlPoint2.x, controlPoint2.y,
            end.x, end.y
        )
    }
    
    drawPath(
        path = path,
        color = Color.Gray,
        style = Stroke(width = 2.dp.toPx())
    )
}
```

## Day 9: 工作流执行引擎

### 9.1 执行引擎实现
```kotlin
fun executeWorkflow(onResult: (String) -> Unit) {
    viewModelScope.launch {
        try {
            val result = StringBuilder()
            result.appendLine("🚀 开始执行工作流: ${_workflow.value.name}")
            result.appendLine("⏰ 执行时间: ${SimpleDateFormat("HH:mm:ss").format(System.currentTimeMillis())}")
            result.appendLine()
            
            val executedNodes = mutableSetOf<String>()
            val startNode = _workflow.value.nodes.find { it.type == NodeType.START }
            
            if (startNode != null) {
                executeNode(startNode, result, executedNodes)
            }
            
            result.appendLine("✅ 工作流执行完成")
            onResult(result.toString())
            
        } catch (e: Exception) {
            onResult("❌ 执行失败: ${e.message}")
        }
    }
}

private suspend fun executeNode(
    node: WorkflowNode, 
    result: StringBuilder, 
    executed: MutableSet<String>
) {
    if (node.id in executed) return
    executed.add(node.id)
    
    result.appendLine("📍 执行节点: ${node.title} (${node.type.displayName})")
    
    when (node.type) {
        NodeType.START -> {
            result.appendLine("   ▶️ 工作流启动")
        }
        NodeType.HTTP_REQUEST -> {
            val url = node.config["url"] as? String ?: "未配置URL"
            result.appendLine("   🌐 请求URL: $url")
            
            try {
                val request = Request.Builder().url(url).build()
                val response = httpClient.newCall(request).execute()
                val responseBody = response.body?.string() ?: ""
                
                result.appendLine("   📊 状态码: ${response.code}")
                result.appendLine("   📄 响应长度: ${responseBody.length} 字符")
                if (responseBody.length < 200) {
                    result.appendLine("   📝 响应内容: ${responseBody.take(100)}...")
                }
            } catch (e: Exception) {
                result.appendLine("   ❌ 请求失败: ${e.message}")
            }
        }
        NodeType.END -> {
            result.appendLine("   🏁 工作流结束")
        }
        else -> {
            result.appendLine("   ⚙️ 节点处理完成")
        }
    }
    
    result.appendLine()
    
    // 执行下一个连接的节点
    val connections = _workflow.value.connections.filter { it.sourceNodeId == node.id }
    connections.forEach { connection ->
        val nextNode = _workflow.value.nodes.find { it.id == connection.targetNodeId }
        if (nextNode != null) {
            executeNode(nextNode, result, executed)
        }
    }
}
```

```kotlin
@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun WorkflowEditor(
    viewModel: WorkflowViewModel = viewModel(),
    modifier: Modifier = Modifier
) {
    val workflow by viewModel.workflow.collectAsState()
    val selectedNodeId by viewModel.selectedNodeId.collectAsState()
    
    var showNodePalette by remember { mutableStateOf(false) }
    var showImportDialog by remember { mutableStateOf(false) }
    var showExportDialog by remember { mutableStateOf(false) }
    var showExecuteDialog by remember { mutableStateOf(false) }
    var connectingFrom by remember { mutableStateOf<Pair<String, String>?>(null) }

    Column(modifier = modifier.fillMaxSize()) {
        // 工具栏
        TopAppBar(
            title = { 
                Column {
                    Text(workflow.name)
                    Text(
                        text = "节点: ${workflow.nodes.size}, 连接: ${workflow.connections.size}",
                        fontSize = 10.sp,
                        color = MaterialTheme.colorScheme.onSurfaceVariant
                    )
                }
            },
            actions = {
                IconButton(onClick = { showNodePalette = true }) {
                    Icon(Icons.Default.Add, "添加节点")
                }
                IconButton(onClick = { showImportDialog = true }) {
                    Icon(Icons.Default.FileOpen, "导入")
                }
                IconButton(onClick = { viewModel.loadSampleWorkflow() }) {
                    Icon(Icons.Default.Lightbulb, "示例")
                }
                IconButton(onClick = { showExportDialog = true }) {
                    Icon(Icons.Default.Save, "导出")
                }
                IconButton(onClick = { 
                    viewModel.executeWorkflow { result ->
                        showExecuteDialog = true
                    }
                }) {
                    Icon(Icons.Default.PlayArrow, "执行")
                }
            }
        )

        // 连接状态提示
        if (connectingFrom != null) {
            Card(
                modifier = Modifier
                    .fillMaxWidth()
                    .padding(8.dp),
                colors = CardDefaults.cardColors(
                    containerColor = MaterialTheme.colorScheme.primaryContainer
                )
            ) {
                Text(
                    text = "连接模式：点击目标节点的输入端口完成连接",
                    modifier = Modifier.padding(12.dp),
                    color = MaterialTheme.colorScheme.onPrimaryContainer
                )
            }
        }

        // 画布区域
        Box(
            modifier = Modifier
                .fillMaxSize()
                .background(Color(0xFFF5F5F5))
                .pointerInput(Unit) {
                    detectTapGestures { 
                        viewModel.selectNode(null)
                        connectingFrom = null
                    }
                }
        ) {
            // 渲染连接线
            workflow.connections.forEach { connection ->
                val sourceNode = workflow.nodes.find { it.id == connection.sourceNodeId }
                val targetNode = workflow.nodes.find { it.id == connection.targetNodeId }
                
                if (sourceNode != null && targetNode != null) {
                    ConnectionLine(
                        startX = sourceNode.x + 160f,
                        startY = sourceNode.y + 50f,
                        endX = targetNode.x,
                        endY = targetNode.y + 50f
                    )
                }
            }

            // 渲染节点
            workflow.nodes.forEach { node ->
                WorkflowNodeView(
                    node = node,
                    isSelected = node.id == selectedNodeId,
                    connectingFrom = connectingFrom,
                    onMove = { deltaX, deltaY ->
                        viewModel.moveNode(node.id, deltaX, deltaY)
                    },
                    onSelect = { viewModel.selectNode(node.id) },
                    onDelete = { viewModel.deleteNode(node.id) },
                    onStartConnection = { outputId ->
                        connectingFrom = node.id to outputId
                    },
                    onCompleteConnection = { inputId ->
                        connectingFrom?.let { (sourceNodeId, outputId) ->
                            viewModel.addConnection(sourceNodeId, outputId, node.id, inputId)
                        }
                        connectingFrom = null
                    },
                    modifier = Modifier.offset(
                        x = with(LocalDensity.current) { node.x.dp },
                        y = with(LocalDensity.current) { node.y.dp }
                    )
                )
            }
        }
    }

    // 对话框处理
    if (showNodePalette) {
        NodePaletteDialog(
            onDismiss = { showNodePalette = false },
            onNodeSelected = { nodeType ->
                viewModel.addNode(nodeType, 100f, 100f)
                showNodePalette = false
            }
        )
    }

    if (showImportDialog) {
        ImportDialog(
            onDismiss = { showImportDialog = false },
            onImport = { json ->
                viewModel.importFromJson(json)
                showImportDialog = false
            }
        )
    }

    if (showExportDialog) {
        ExportDialog(
            json = viewModel.exportToJson(),
            onDismiss = { showExportDialog = false }
        )
    }

    if (showExecuteDialog) {
        var executeResult by remember { mutableStateOf("") }
        
        LaunchedEffect(showExecuteDialog) {
            if (showExecuteDialog) {
                viewModel.executeWorkflow { result ->
                    executeResult = result
                }
            }
        }
        
        ExecuteResultDialog(
            result = executeResult,
            onDismiss = { showExecuteDialog = false }
        )
    }
}
```
