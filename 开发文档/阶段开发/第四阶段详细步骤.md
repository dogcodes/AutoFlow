```kotlin
// WechatRedPacketDetector.kt
class WechatRedPacketDetector {
    
    private val accessibilityHelper = AccessibilityHelper()
    
    // 检测是否在微信界面
    fun isInWechatApp(): Boolean {
        val service = AutoFlowAccessibilityService.getInstance() ?: return false
        val rootNode = service.rootInActiveWindow ?: return false
        return rootNode.packageName == WechatConstants.WECHAT_PACKAGE
    }
    
    // 检测红包是否出现
    fun detectRedPacket(): RedPacketInfo? {
        if (!isInWechatApp()) return null
        
        val service = AutoFlowAccessibilityService.getInstance() ?: return null
        val rootNode = service.rootInActiveWindow ?: return null
        
        // 查找红包元素
        val redPacketNodes = rootNode.findAccessibilityNodeInfosByText(WechatConstants.RED_PACKET_TEXT)
        
        return redPacketNodes?.firstOrNull()?.let { node ->
            RedPacketInfo(
                node = node,
                text = node.text?.toString() ?: "",
                bounds = android.graphics.Rect().apply { node.getBoundsInScreen(this) },
                isClickable = node.isClickable,
                timestamp = System.currentTimeMillis()
            )
        }
    }
    
    // 检测红包详情页面
    fun isRedPacketDetailPage(): Boolean {
        val service = AutoFlowAccessibilityService.getInstance() ?: return false
        val rootNode = service.rootInActiveWindow ?: return false
        
        // 查找"開"按钮
        val openButton = rootNode.findAccessibilityNodeInfosByText(WechatConstants.OPEN_RED_PACKET_TEXT)
        return openButton?.isNotEmpty() == true
    }
    
    // 检测红包状态
    fun getRedPacketStatus(): RedPacketStatus {
        val service = AutoFlowAccessibilityService.getInstance() ?: return RedPacketStatus.UNKNOWN
        val rootNode = service.rootInActiveWindow ?: return RedPacketStatus.UNKNOWN
        
        return when {
            rootNode.findAccessibilityNodeInfosByText(WechatConstants.RED_PACKET_OPENED_TEXT)?.isNotEmpty() == true -> 
                RedPacketStatus.ALREADY_OPENED
            rootNode.findAccessibilityNodeInfosByText(WechatConstants.RED_PACKET_EXPIRED_TEXT)?.isNotEmpty() == true -> 
                RedPacketStatus.EXPIRED
            rootNode.findAccessibilityNodeInfosByText(WechatConstants.OPEN_RED_PACKET_TEXT)?.isNotEmpty() == true -> 
                RedPacketStatus.AVAILABLE
            else -> RedPacketStatus.UNKNOWN
        }
    }
}

data class RedPacketInfo(
    val node: AccessibilityNodeInfo,
    val text: String,
    val bounds: android.graphics.Rect,
    val isClickable: Boolean,
    val timestamp: Long
)

enum class RedPacketStatus {
    AVAILABLE,      // 可抢
    ALREADY_OPENED, // 已被抢完
    EXPIRED,        // 已过期
    UNKNOWN         // 未知状态
}
```

## Day 21: 抢红包自动化逻辑

### 21.1 抢红包执行器
```kotlin
// WechatRedPacketAutomation.kt
class WechatRedPacketAutomation {
    
    private val detector = WechatRedPacketDetector()
    private val accessibilityHelper = AccessibilityHelper()
    private val statistics = RedPacketStatistics()
    
    // 执行抢红包流程
    suspend fun executeRedPacketFlow(config: RedPacketConfig): RedPacketResult {
        val result = RedPacketResult()
        
        try {
            // 1. 检测红包
            val redPacket = detector.detectRedPacket()
            if (redPacket == null) {
                result.success = false
                result.message = "未检测到红包"
                return result
            }
            
            result.detectionTime = System.currentTimeMillis() - redPacket.timestamp
            
            // 2. 随机延时（避免检测）
            val randomDelay = config.minDelay + (Math.random() * (config.maxDelay - config.minDelay)).toLong()
            delay(randomDelay)
            
            // 3. 点击红包
            val clickSuccess = accessibilityHelper.performClick("text=${WechatConstants.RED_PACKET_TEXT}")
            if (!clickSuccess) {
                result.success = false
                result.message = "点击红包失败"
                return result
            }
            
            // 4. 等待红包详情页面
            val detailPageAppeared = accessibilityHelper.waitForElement(
                "text=${WechatConstants.OPEN_RED_PACKET_TEXT}", 
                config.waitTimeout
            )
            
            if (!detailPageAppeared) {
                result.success = false
                result.message = "红包详情页面未出现"
                return result
            }
            
            // 5. 检查红包状态
            val status = detector.getRedPacketStatus()
            when (status) {
                RedPacketStatus.AVAILABLE -> {
                    // 6. 点击开红包
                    val openSuccess = accessibilityHelper.performClick("text=${WechatConstants.OPEN_RED_PACKET_TEXT}")
                    if (openSuccess) {
                        result.success = true
                        result.message = "抢红包成功"
                        
                        // 7. 尝试获取金额（可选）
                        delay(1000)
                        result.amount = extractRedPacketAmount()
                    } else {
                        result.success = false
                        result.message = "点击开红包失败"
                    }
                }
                RedPacketStatus.ALREADY_OPENED -> {
                    result.success = false
                    result.message = "红包已被抢完"
                }
                RedPacketStatus.EXPIRED -> {
                    result.success = false
                    result.message = "红包已过期"
                }
                else -> {
                    result.success = false
                    result.message = "红包状态未知"
                }
            }
            
            // 8. 返回聊天界面
            delay(config.returnDelay)
            performBackAction()
            
        } catch (e: Exception) {
            result.success = false
            result.message = "执行异常: ${e.message}"
        }
        
        // 记录统计信息
        statistics.recordAttempt(result)
        
        return result
    }
    
    // 提取红包金额
    private fun extractRedPacketAmount(): Double? {
        val service = AutoFlowAccessibilityService.getInstance() ?: return null
        val rootNode = service.rootInActiveWindow ?: return null
        
        // 查找金额相关的文本节点
        val amountPattern = Regex("\\d+\\.\\d{2}")
        
        return findAmountInNode(rootNode, amountPattern)
    }
    
    private fun findAmountInNode(node: AccessibilityNodeInfo, pattern: Regex): Double? {
        // 检查当前节点的文本
        node.text?.toString()?.let { text ->
            pattern.find(text)?.value?.toDoubleOrNull()?.let { return it }
        }
        
        // 递归检查子节点
        for (i in 0 until node.childCount) {
            val child = node.getChild(i)
            child?.let {
                findAmountInNode(it, pattern)?.let { amount -> return amount }
            }
        }
        
        return null
    }
    
    // 执行返回操作
    private fun performBackAction() {
        val service = AutoFlowAccessibilityService.getInstance()
        service?.performGlobalAction(AccessibilityService.GLOBAL_ACTION_BACK)
    }
}

data class RedPacketConfig(
    val minDelay: Long = 100,      // 最小延时(ms)
    val maxDelay: Long = 500,      // 最大延时(ms)
    val waitTimeout: Long = 3000,  // 等待超时(ms)
    val returnDelay: Long = 2000,  // 返回延时(ms)
    val enableStatistics: Boolean = true,  // 启用统计
    val maxAttemptsPerHour: Int = 50      // 每小时最大尝试次数
)

data class RedPacketResult(
    var success: Boolean = false,
    var message: String = "",
    var amount: Double? = null,
    var detectionTime: Long = 0,
    val timestamp: Long = System.currentTimeMillis()
)
```

```kotlin
// RedPacketStatistics.kt
class RedPacketStatistics {
    
    private val _statistics = MutableStateFlow(StatisticsData())
    val statistics: StateFlow<StatisticsData> = _statistics
    
    fun recordAttempt(result: RedPacketResult) {
        val current = _statistics.value
        
        _statistics.value = current.copy(
            totalAttempts = current.totalAttempts + 1,
            successfulAttempts = if (result.success) current.successfulAttempts + 1 else current.successfulAttempts,
            totalAmount = current.totalAmount + (result.amount ?: 0.0),
            lastAttemptTime = result.timestamp,
            averageDetectionTime = calculateAverageDetectionTime(current, result.detectionTime)
        )
    }
    
    fun getHourlyAttempts(): Int {
        val oneHourAgo = System.currentTimeMillis() - 3600000 // 1小时
        // 这里需要实现基于时间的统计逻辑
        return 0 // 简化实现
    }
    
    fun resetStatistics() {
        _statistics.value = StatisticsData()
    }
    
    private fun calculateAverageDetectionTime(current: StatisticsData, newTime: Long): Long {
        return if (current.totalAttempts == 0) {
            newTime
        } else {
            (current.averageDetectionTime * current.totalAttempts + newTime) / (current.totalAttempts + 1)
        }
    }
}

data class StatisticsData(
    val totalAttempts: Int = 0,
    val successfulAttempts: Int = 0,
    val totalAmount: Double = 0.0,
    val lastAttemptTime: Long = 0,
    val averageDetectionTime: Long = 0
) {
    val successRate: Double
        get() = if (totalAttempts > 0) successfulAttempts.toDouble() / totalAttempts else 0.0
}
```

```kotlin
@Composable
private fun StatisticItem(label: String, value: String) {
    Column(horizontalAlignment = Alignment.CenterHorizontally) {
        Text(
            text = value,
            style = MaterialTheme.typography.titleMedium,
            fontWeight = FontWeight.Bold,
            color = MaterialTheme.colorScheme.primary
        )
        Text(
            text = label,
            style = MaterialTheme.typography.bodySmall,
            color = MaterialTheme.colorScheme.onSurfaceVariant
        )
    }
}

private fun formatTime(timestamp: Long): String {
    if (timestamp == 0L) return "无"
    val format = SimpleDateFormat("HH:mm", Locale.getDefault())
    return format.format(Date(timestamp))
}
```
