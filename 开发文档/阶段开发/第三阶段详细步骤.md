// AutoFlowAccessibilityService.kt
class AutoFlowAccessibilityService : AccessibilityService() {
    
    companion object {
        private var instance: AutoFlowAccessibilityService? = null
        
        fun getInstance(): AutoFlowAccessibilityService? = instance
        
        fun isServiceEnabled(): Boolean = instance != null
    }

    override fun onServiceConnected() {
        super.onServiceConnected()
        instance = this
        Log.d("AccessibilityService", "服务已连接")
    }

    override fun onDestroy() {
        super.onDestroy()
        instance = null
        Log.d("AccessibilityService", "服务已断开")
    }

    override fun onAccessibilityEvent(event: AccessibilityEvent?) {
        // 处理无障碍事件
        event?.let { 
            handleAccessibilityEvent(it)
        }
    }

    override fun onInterrupt() {
        Log.d("AccessibilityService", "服务被中断")
    }

    private fun handleAccessibilityEvent(event: AccessibilityEvent) {
        when (event.eventType) {
            AccessibilityEvent.TYPE_WINDOW_STATE_CHANGED -> {
                // 窗口状态变化
                Log.d("AccessibilityService", "窗口变化: ${event.packageName}")
            }
            AccessibilityEvent.TYPE_VIEW_CLICKED -> {
                // 视图点击事件
                Log.d("AccessibilityService", "点击事件: ${event.text}")
            }
        }
    }

    // 查找节点
    fun findNodeByText(text: String): AccessibilityNodeInfo? {
        return rootInActiveWindow?.findAccessibilityNodeInfosByText(text)?.firstOrNull()
    }

    fun findNodeById(resourceId: String): AccessibilityNodeInfo? {
        return rootInActiveWindow?.findAccessibilityNodeInfosByViewId(resourceId)?.firstOrNull()
    }

    // 执行点击
    fun performClick(node: AccessibilityNodeInfo): Boolean {
        return node.performAction(AccessibilityNodeInfo.ACTION_CLICK)
    }

    // 输入文本
    fun performInput(node: AccessibilityNodeInfo, text: String): Boolean {
        val arguments = Bundle().apply {
            putCharSequence(AccessibilityNodeInfo.ACTION_ARGUMENT_SET_TEXT_CHARSEQUENCE, text)
        }
        return node.performAction(AccessibilityNodeInfo.ACTION_SET_TEXT, arguments)
    }

    // 执行滚动
    fun performScroll(direction: String): Boolean {
        val action = when (direction) {
            "up" -> AccessibilityNodeInfo.ACTION_SCROLL_BACKWARD
            "down" -> AccessibilityNodeInfo.ACTION_SCROLL_FORWARD
            else -> return false
        }
        return rootInActiveWindow?.performAction(action) ?: false
    }
}

## Day 15: 无障碍操作执行器

### 15.1 创建AccessibilityHelper
```kotlin
// AccessibilityHelper.kt
class AccessibilityHelper {
    
    companion object {
        fun isAccessibilityEnabled(context: Context): Boolean {
            val accessibilityManager = context.getSystemService(Context.ACCESSIBILITY_SERVICE) as AccessibilityManager
            val enabledServices = Settings.Secure.getString(
                context.contentResolver,
                Settings.Secure.ENABLED_ACCESSIBILITY_SERVICES
            )
            return enabledServices?.contains(context.packageName) == true
        }
        
        fun openAccessibilitySettings(context: Context) {
            val intent = Intent(Settings.ACTION_ACCESSIBILITY_SETTINGS)
            context.startActivity(intent)
        }
    }
    
    private val service: AutoFlowAccessibilityService?
        get() = AutoFlowAccessibilityService.getInstance()

    // 元素选择器解析
    fun parseSelector(selector: String): ElementSelector {
        return when {
            selector.startsWith("text=") -> ElementSelector.ByText(selector.substring(5))
            selector.startsWith("id=") -> ElementSelector.ById(selector.substring(3))
            selector.startsWith("class=") -> ElementSelector.ByClass(selector.substring(6))
            selector.startsWith("desc=") -> ElementSelector.ByDescription(selector.substring(5))
            else -> ElementSelector.ByText(selector)
        }
    }

    // 查找元素
    fun findElement(selector: ElementSelector): AccessibilityNodeInfo? {
        val service = this.service ?: return null
        
        return when (selector) {
            is ElementSelector.ByText -> service.findNodeByText(selector.text)
            is ElementSelector.ById -> service.findNodeById(selector.id)
            is ElementSelector.ByClass -> findNodeByClassName(selector.className)
            is ElementSelector.ByDescription -> findNodeByDescription(selector.description)
        }
    }

    // 执行点击操作
    fun performClick(selector: String, clickType: String = "single"): Boolean {
        val elementSelector = parseSelector(selector)
        val node = findElement(elementSelector) ?: return false
        
        return when (clickType) {
            "single" -> service?.performClick(node) ?: false
            "double" -> {
                service?.performClick(node)
                Thread.sleep(100)
                service?.performClick(node) ?: false
            }
            "long" -> node.performAction(AccessibilityNodeInfo.ACTION_LONG_CLICK)
            else -> false
        }
    }

    // 执行输入操作
    fun performInput(selector: String, text: String, clearFirst: Boolean = true): Boolean {
        val elementSelector = parseSelector(selector)
        val node = findElement(elementSelector) ?: return false
        
        if (clearFirst) {
            node.performAction(AccessibilityNodeInfo.ACTION_SELECT_ALL)
        }
        
        return service?.performInput(node, text) ?: false
    }

    // 等待元素出现
    suspend fun waitForElement(selector: String, timeout: Long = 10000): Boolean {
        val startTime = System.currentTimeMillis()
        val elementSelector = parseSelector(selector)
        
        while (System.currentTimeMillis() - startTime < timeout) {
            if (findElement(elementSelector) != null) {
                return true
            }
            delay(500) // 每500ms检查一次
        }
        
        return false
    }

    // 执行滚动
    fun performScroll(direction: String, distance: Int = 1): Boolean {
        return service?.performScroll(direction) ?: false
    }

    // 检查元素属性
    fun checkElement(selector: String, checkType: String, expectedValue: String): Boolean {
        val elementSelector = parseSelector(selector)
        val node = findElement(elementSelector) ?: return false
        
        return when (checkType) {
            "text" -> node.text?.toString() == expectedValue
            "enabled" -> node.isEnabled == expectedValue.toBoolean()
            "visible" -> node.isVisibleToUser == expectedValue.toBoolean()
            "exists" -> true // 如果找到了节点就说明存在
            else -> false
        }
    }

    private fun findNodeByClassName(className: String): AccessibilityNodeInfo? {
        // 实现按类名查找的逻辑
        return service?.rootInActiveWindow?.let { root ->
            findNodeByClassNameRecursive(root, className)
        }
    }

    private fun findNodeByDescription(description: String): AccessibilityNodeInfo? {
        // 实现按描述查找的逻辑
        return service?.rootInActiveWindow?.let { root ->
            findNodeByDescriptionRecursive(root, description)
        }
    }

    private fun findNodeByClassNameRecursive(node: AccessibilityNodeInfo, className: String): AccessibilityNodeInfo? {
        if (node.className?.toString() == className) {
            return node
        }
        
        for (i in 0 until node.childCount) {
            val child = node.getChild(i)
            child?.let {
                val result = findNodeByClassNameRecursive(it, className)
                if (result != null) return result
            }
        }
        
        return null
    }

    private fun findNodeByDescriptionRecursive(node: AccessibilityNodeInfo, description: String): AccessibilityNodeInfo? {
        if (node.contentDescription?.toString() == description) {
            return node
        }
        
        for (i in 0 until node.childCount) {
            val child = node.getChild(i)
            child?.let {
                val result = findNodeByDescriptionRecursive(it, description)
                if (result != null) return result
            }
        }
        
        return null
    }
}

sealed class ElementSelector {
    data class ByText(val text: String) : ElementSelector()
    data class ById(val id: String) : ElementSelector()
    data class ByClass(val className: String) : ElementSelector()
    data class ByDescription(val description: String) : ElementSelector()
}
```

private suspend fun executeNode(
    node: WorkflowNode, 
    result: StringBuilder, 
    executed: MutableSet<String>
) {
    if (node.id in executed) return
    executed.add(node.id)
    
    result.appendLine("📍 执行节点: ${node.title} (${node.type.displayName})")
    
    when (node.type) {
        // 现有节点处理...
        
        NodeType.ACCESSIBILITY_CLICK -> {
            val selector = node.config["selector"] as? String ?: ""
            val clickType = node.config["clickType"] as? String ?: "single"
            
            result.appendLine("   👆 点击操作: $selector")
            
            if (!AutoFlowAccessibilityService.isServiceEnabled()) {
                result.appendLine("   ❌ 无障碍服务未启用")
                return
            }
            
            try {
                val success = AccessibilityHelper().performClick(selector, clickType)
                if (success) {
                    result.appendLine("   ✅ 点击成功")
                } else {
                    result.appendLine("   ❌ 点击失败：未找到元素")
                }
            } catch (e: Exception) {
                result.appendLine("   ❌ 点击失败: ${e.message}")
            }
        }
        
        NodeType.ACCESSIBILITY_INPUT -> {
            val selector = node.config["selector"] as? String ?: ""
            val text = node.config["text"] as? String ?: ""
            val clearFirst = node.config["clearFirst"] as? Boolean ?: true
            
            result.appendLine("   ⌨️ 输入操作: $selector -> $text")
            
            if (!AutoFlowAccessibilityService.isServiceEnabled()) {
                result.appendLine("   ❌ 无障碍服务未启用")
                return
            }
            
            try {
                val success = AccessibilityHelper().performInput(selector, text, clearFirst)
                if (success) {
                    result.appendLine("   ✅ 输入成功")
                } else {
                    result.appendLine("   ❌ 输入失败：未找到输入框")
                }
            } catch (e: Exception) {
                result.appendLine("   ❌ 输入失败: ${e.message}")
            }
        }
        
        NodeType.ACCESSIBILITY_WAIT -> {
            val selector = node.config["selector"] as? String ?: ""
            val timeout = (node.config["timeout"] as? Number)?.toLong() ?: 10000L
            
            result.appendLine("   ⏳ 等待元素: $selector (超时: ${timeout}ms)")
            
            if (!AutoFlowAccessibilityService.isServiceEnabled()) {
                result.appendLine("   ❌ 无障碍服务未启用")
                return
            }
            
            try {
                val found = AccessibilityHelper().waitForElement(selector, timeout)
                if (found) {
                    result.appendLine("   ✅ 元素已出现")
                } else {
                    result.appendLine("   ❌ 等待超时：元素未出现")
                }
            } catch (e: Exception) {
                result.appendLine("   ❌ 等待失败: ${e.message}")
            }
        }
        
        NodeType.ACCESSIBILITY_SCROLL -> {
            val direction = node.config["direction"] as? String ?: "down"
            val distance = (node.config["distance"] as? Number)?.toInt() ?: 1
            
            result.appendLine("   📜 滚动操作: $direction")
            
            if (!AutoFlowAccessibilityService.isServiceEnabled()) {
                result.appendLine("   ❌ 无障碍服务未启用")
                return
            }
            
            try {
                val success = AccessibilityHelper().performScroll(direction, distance)
                if (success) {
                    result.appendLine("   ✅ 滚动成功")
                } else {
                    result.appendLine("   ❌ 滚动失败")
                }
            } catch (e: Exception) {
                result.appendLine("   ❌ 滚动失败: ${e.message}")
            }
        }
        
        NodeType.ACCESSIBILITY_CHECK -> {
            val selector = node.config["selector"] as? String ?: ""
            val checkType = node.config["checkType"] as? String ?: "exists"
            val expectedValue = node.config["expectedValue"] as? String ?: ""
            
            result.appendLine("   🔍 检查元素: $selector ($checkType)")
            
            if (!AutoFlowAccessibilityService.isServiceEnabled()) {
                result.appendLine("   ❌ 无障碍服务未启用")
                return
            }
            
            try {
                val checkResult = AccessibilityHelper().checkElement(selector, checkType, expectedValue)
                if (checkResult) {
                    result.appendLine("   ✅ 检查通过")
                } else {
                    result.appendLine("   ❌ 检查失败")
                }
            } catch (e: Exception) {
                result.appendLine("   ❌ 检查失败: ${e.message}")
            }
        }
        
        else -> {
            result.appendLine("   ⚙️ 节点处理完成")
        }
    }
    
    result.appendLine()
    
    // 执行下一个连接的节点
    val connections = _workflow.value.connections.filter { it.sourceNodeId == node.id }
    connections.forEach { connection ->
        val nextNode = _workflow.value.nodes.find { it.id == connection.targetNodeId }
        if (nextNode != null) {
            executeNode(nextNode, result, executed)
        }
    }
}

@Composable
fun AccessibilityPermissionCard(
    modifier: Modifier = Modifier
) {
    val context = LocalContext.current
    var isEnabled by remember { mutableStateOf(AccessibilityHelper.isAccessibilityEnabled(context)) }
    
    LaunchedEffect(Unit) {
        // 定期检查权限状态
        while (true) {
            isEnabled = AccessibilityHelper.isAccessibilityEnabled(context)
            delay(1000)
        }
    }
    
    Card(
        modifier = modifier
            .fillMaxWidth()
            .padding(16.dp),
        colors = CardDefaults.cardColors(
            containerColor = if (isEnabled) 
                MaterialTheme.colorScheme.primaryContainer
            else MaterialTheme.colorScheme.errorContainer
        )
    ) {
        Column(
            modifier = Modifier.padding(16.dp)
        ) {
            Row(
                verticalAlignment = Alignment.CenterVertically
            ) {
                Icon(
                    imageVector = if (isEnabled) Icons.Default.CheckCircle else Icons.Default.Warning,
                    contentDescription = null,
                    tint = if (isEnabled) 
                        MaterialTheme.colorScheme.onPrimaryContainer
                    else MaterialTheme.colorScheme.onErrorContainer

                Spacer(modifier = Modifier.width(8.dp))
                Text(
                    text = if (isEnabled) "无障碍服务已启用" else "无障碍服务未启用",

                }
            }
        },
        confirmButton = {
            TextButton(onClick = onDismiss) {
                Text("关闭")
            }
        }
    )
}

private fun inspectCurrentScreen(): String {
    val service = AutoFlowAccessibilityService.getInstance()
    if (service == null) {
        return "无障碍服务未启用"
    }
    
    val rootNode = service.rootInActiveWindow
    if (rootNode == null) {
        return "无法获取屏幕信息"
    }
    
    val result = StringBuilder()
    result.appendLine("屏幕元素信息:")
    result.appendLine("包名: ${rootNode.packageName}")
    result.appendLine("类名: ${rootNode.className}")
    result.appendLine()
    
    inspectNodeRecursive(rootNode, result, 0)
    
    return result.toString()
}

private fun inspectNodeRecursive(node: AccessibilityNodeInfo, result: StringBuilder, depth: Int) {
    val indent = "  ".repeat(depth)
    
    result.appendLine("${indent}节点信息:")
    result.appendLine("${indent}  类名: ${node.className}")
    result.appendLine("${indent}  文本: ${node.text}")
    result.appendLine("${indent}  描述: ${node.contentDescription}")
    result.appendLine("${indent}  资源ID: ${node.viewIdResourceName}")
    result.appendLine("${indent}  可点击: ${node.isClickable}")
    result.appendLine("${indent}  可输入: ${node.isEditable}")
    result.appendLine()
    
    // 递归检查子节点（限制深度避免输出过多）
    if (depth < 3) {
        for (i in 0 until node.childCount) {
            val child = node.getChild(i)
            child?.let {
                inspectNodeRecursive(it, result, depth + 1)
            }
        }
    }
}
