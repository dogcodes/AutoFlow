### 核心实现架构
```kotlin
// CanvasViewModel.kt
class CanvasViewModel : ViewModel() {
    private val _canvasState = MutableStateFlow(CanvasState())
    val canvasState: StateFlow<CanvasState> = _canvasState
    
    fun updateScale(scale: Float) {
        _canvasState.value = _canvasState.value.copy(
            scale = scale.coerceIn(
                _canvasState.value.minScale,
                _canvasState.value.maxScale
            )
        )
    }
    
    fun updateOffset(offsetX: Float, offsetY: Float) {
        _canvasState.value = _canvasState.value.copy(
            offsetX = offsetX,
            offsetY = offsetY
        )
    }
    
    fun resetZoom() {
        _canvasState.value = CanvasState()
    }
    
    fun zoomIn() {
        updateScale(_canvasState.value.scale * 1.2f)
    }
    
    fun zoomOut() {
        updateScale(_canvasState.value.scale / 1.2f)
    }
    
    fun fitToScreen(canvasSize: Size, contentBounds: Rect) {
        // 计算合适的缩放比例和居中偏移
        val scaleX = canvasSize.width / contentBounds.width
        val scaleY = canvasSize.height / contentBounds.height
        val scale = minOf(scaleX, scaleY, 1f)
        
        val offsetX = (canvasSize.width - contentBounds.width * scale) / 2
        val offsetY = (canvasSize.height - contentBounds.height * scale) / 2
        
        _canvasState.value = _canvasState.value.copy(
            scale = scale,
            offsetX = offsetX,
            offsetY = offsetY
        )
    }
}

// 增强的WorkflowEditor
@Composable
fun WorkflowEditor(
    workflowViewModel: WorkflowViewModel = viewModel(),
    canvasViewModel: CanvasViewModel = viewModel()
) {
    val canvasState by canvasViewModel.canvasState.collectAsState()
    
    Box(modifier = Modifier.fillMaxSize()) {
        // 可缩放的画布
        Box(
            modifier = Modifier
                .fillMaxSize()
                .pointerInput(Unit) {
                    detectTransformGestures { _, pan, zoom, _ ->
                        // 处理缩放
                        canvasViewModel.updateScale(canvasState.scale * zoom)
                        
                        // 处理平移
                        canvasViewModel.updateOffset(
                            canvasState.offsetX + pan.x,
                            canvasState.offsetY + pan.y
                        )
                    }
                }
                .graphicsLayer(
                    scaleX = canvasState.scale,
                    scaleY = canvasState.scale,
                    translationX = canvasState.offsetX,
                    translationY = canvasState.offsetY
                )
        ) {
            // 工作流节点内容
            WorkflowCanvas(workflowViewModel)
        }
        
        // 缩放控制栏 (右下角)
        ZoomControls(
            scale = canvasState.scale,
            onZoomIn = canvasViewModel::zoomIn,
            onZoomOut = canvasViewModel::zoomOut,
            onResetZoom = canvasViewModel::resetZoom,
            modifier = Modifier.align(Alignment.BottomEnd)
        )
        
        // 适应屏幕按钮 (左下角)
        FloatingActionButton(
            onClick = { 
                // 计算所有节点的边界并适应屏幕
                canvasViewModel.fitToScreen(canvasSize, nodesBounds)
            },
            modifier = Modifier
                .align(Alignment.BottomStart)
                .padding(16.dp),
            containerColor = MaterialTheme.colorScheme.secondary
        ) {
            Icon(Icons.Default.FitScreen, "适应屏幕")
        }
    }
}
```
